-- 1. Se a procedure já existir, remove para recriar
IF OBJECT_ID('dbo.st_InsereItensVenda', 'P') IS NOT NULL
    DROP PROCEDURE dbo.st_InsereItensVenda;
GO

-- 2. Cria a procedure (deve ser o primeiro comando do lote após o GO)
CREATE PROCEDURE st_InsereItensVenda   
    @nm_Prod  VARCHAR(50),                                     
    @qtdVenda INTEGER,
    @codVEnda INTEGER,
    @erMsg    VARCHAR(255) OUTPUT, 
    @return   INTEGER      OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @idProd  INTEGER,		  
            @vlVenda MONEY,
            @estoque INTEGER;

    SET @return = 1;                                           
    SET @erMsg = '';
  
    -- Busca dados do produto usando a coluna correta (QTD_PROD)
    SELECT 
        @idProd  = ID_PROD,
        @vlVenda = VL_PROD,
        @estoque = QTD_PROD
    FROM dbo.TBPRODUTOS WITH (NOLOCK)
    WHERE NOME_PROD = @nm_Prod;

    -- Validações
    IF (@idProd IS NULL)
    BEGIN
        SET @return = 2;
        SET @erMsg = 'Produto não encontrado: ' + @nm_Prod;
        RETURN;
    END

    IF (@estoque <= 0)
    BEGIN
        SET @return = 2;
        SET @erMsg = 'Produto ' + @nm_Prod + ' em Falta';
        RETURN;
    END
  
    IF (@qtdVenda > @estoque)
    BEGIN
        SET @return = 2;
        SET @erMsg = 'Estoque Indisponível. Qtd vendida maior que estoque: ' + @nm_Prod + '. Estoque atual = ' + CAST(@estoque AS VARCHAR(10)); 
        RETURN;
    END

    -- Insere o item e atualiza estoque
    INSERT INTO TBITENSVENDA (ID_Cod_Venda, id_prod, vl_prod, QTD_Prod)
    VALUES (@codVEnda, @idProd, @vlVenda, @qtdVenda);   

    UPDATE dbo.TBPRODUTOS
    SET QTD_PROD = @estoque - @qtdVenda
    WHERE ID_PROD = @idProd;

    -- Alerta de estoque baixo
    IF ((@estoque - @qtdVenda) <= 5)
    BEGIN
        SET @return = 3;
        SET @erMsg = 'Estoque Baixo: ' + @nm_Prod;  
    END
END
GO
