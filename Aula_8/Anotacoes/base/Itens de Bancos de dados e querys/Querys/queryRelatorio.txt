SELECT TBVENDAS.ID_Cod_Venda
      ,NOME_CLI
	  ,DT_Venda
	  ,NOME_PROD
	  ,QTD_Prod
	  ,TBITENSVENDA.VL_PROD
	  ,Total
FROM TBITENSVENDA
INNER JOIN TBVENDAS
ON TBVENDAS.ID_Cod_Venda = TBITENSVENDA.ID_Cod_Venda
INNER JOIN TBPRODUTOS
ON TBPRODUTOS.ID_PROD = TBITENSVENDA.ID_Prod
INNER JOIN TBCLIENTES
ON TBCLIENTES.ID_CLI = TBVENDAS.ID_Cli
