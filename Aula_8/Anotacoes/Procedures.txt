 sp_AdicionarCliente 
-------------------------------------------------------------------
CREATE PROCEDURE sp_AdicionarCliente
    @Nome VARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;
    
    INSERT INTO TBCLIENTES (NOME_CLI)
    VALUES (@Nome);
END;
GO
------------------------------------------------------------------- 
  
 sp_ApagarCliente
-------------------------------------------------------------------------------
CREATE PROCEDURE sp_EditarCliente
    @Id INT,
    @Nome VARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;
    
    UPDATE TBCLIENTES
    SET NOME_CLI = @Nome
    WHERE ID_CLI = @Id;
END;
GO
--------------------------------------------------------------------------------

 sp_EditarCliente
-------------------------------------------------------------------------------
CREATE PROCEDURE sp_ApagarCliente
    @Id INT
AS
BEGIN
    SET NOCOUNT ON;
    
    DELETE FROM TBCLIENTES
    WHERE ID_CLI = @Id;
END;
GO
-------------------------------------------------------------------------------

 sp_ListarClientes
-------------------------------------------------------------------------------
CREATE PROCEDURE sp_ListarClientes
AS
BEGIN
    SET NOCOUNT ON;
    
    SELECT ID_CLI, NOME_CLI 
    FROM TBCLIENTES
    ORDER BY NOME_CLI;
END;
GO
--------------------------------------------------------------------------------

procedure TfrmClientes.btnSalvarClick(Sender: TObject);
begin
  with dmDados.DataModule1.stInsereCliente do
  begin
    Close;

    // 1. Força o Delphi a ir buscar os parâmetros no SQL Server
    Parameters.Refresh;

    // 1. Passa o valor do seu TEdit (txtNome) para o parâmetro da procedure
    Parameters.ParamByName('@Nome').Value := txtNome.Text;

    // 2. Executa a procedure no SQL Server
    ExecProc;
  end;
